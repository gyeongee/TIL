### roboFflow
- roboflow에서 데이터 라벨링 진행하여 데이터 생성 후 모델링 진행

  ```
  # Roboflow 라이브러리를 불러옵니다.
from roboflow import Roboflow

# Roboflow 객체 생성 및 API 키로 인증
# "u" 부분에는 본인의 Roboflow API Key를 넣어야 합니다.
rf = Roboflow(api_key="u")

# 특정 워크스페이스 안의 프로젝트 불러오기
# "data-xpqum" : 워크스페이스 이름
# "car-z5zwx-c0t3n" : 프로젝트 이름
project = rf.workspace("data-xpqum").project("car-z5zwx-c0t3n")

# 프로젝트의 특정 버전 선택
# 여기서는 1번 버전을 선택
version = project.version(1)

# 선택한 버전의 데이터셋 다운로드
# "yolov11" 포맷으로 다운로드 (YOLOv11 모델 학습용)
dataset = version.download("yolov11")

# 다운로드된 데이터셋 경로가 dataset 변수에 저장됨
# 이후 YOLO 모델 학습에 바로 사용 가능

  ```
